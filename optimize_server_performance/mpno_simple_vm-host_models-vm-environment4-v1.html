# Creates the rolling windows (without host information data about migrations and reconfs)
# change window size for different levels of datetime aggregation
# remove host-related code for vm-only models


def create_historic_features(df, window_size=7):

    #     sorted_df = df.sort_values(["date", "host", "vm"]).reset_index(drop=True)
    sorted_df = df.sort_values(["vm", "date"]).reset_index(drop=True)
    rolling_df = sorted_df.rolling(window_size)

    # vm performance measures
    sorted_df["guestmemoryusage_week_avr"] = rolling_df.mean()["guestmemoryusage"]
    sorted_df["guestmemoryusage_week_sd"] = rolling_df.std()["guestmemoryusage"]
    #     sorted_df["ballooned_week_avr"] = rolling_df.mean()["ballooned"]
    #     sorted_df["ballooned_week_sd"] = rolling_df.std()["ballooned"]
    sorted_df["compressed_week_avr"] = rolling_df.mean()["compressed"]
    sorted_df["compressed_week_sd"] = rolling_df.std()["compressed"]
    sorted_df["consumedOverhead_week_avr"] = rolling_df.mean()["consumedOverhead"]
    sorted_df["consumedOverhead_week_sd"] = rolling_df.std()["consumedOverhead"]
    sorted_df["cpuusage_week_avr"] = rolling_df.mean()["cpuusage"]
    sorted_df["cpuusage_week_sd"] = rolling_df.std()["cpuusage"]

    #     # host performance measures
    #     sorted_df["overallCpuUsage_host_week_avr"] = rolling_df.mean()[
    #         "overallCpuUsage_host"
    #     ]
    #     sorted_df["overallCpuUsage_host_week_sd"] = rolling_df.std()["overallCpuUsage_host"]
    #     sorted_df["overallMemoryUsage_host_week_avr"] = rolling_df.mean()[
    #         "overallMemoryUsage_host"
    #     ]
    #     sorted_df["overallMemoryUsage_host_week_sd"] = rolling_df.std()[
    #         "overallMemoryUsage_host"
    #     ]
    #     sorted_df["Uptime_host_week_avr"] = rolling_df.mean()["Uptime_host"]
    #     sorted_df["Uptime_host_week_sd"] = rolling_df.std()["Uptime_host"]

    #     df = sorted_df.sort_values(["date", "host", "vm"]).reset_index(drop=True)
    df = sorted_df.sort_values(["date", "vm"]).reset_index(drop=True)
    return df
